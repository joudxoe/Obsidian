# background-position:100% bottom;
# background-position:100% right;

```
# 將圖像置於左上角

# background-position:top left;

# background-position:left top;
# background-position:left 0%;
# background-position:0% 0%;
# background-position:0% left;

```
# 將圖像置於左下角

# background-position:left bottom;

```
# 圖像居中

# background-position:center;
# background-position:center center;
# background-position:50% 50%;
```

```
# 偏移點是圖像的左上角，向右50像素，向下20像素

# background-position:50px 20px;
```

```
# 設定背景顏色

# background-color:#f0e68c;
```

```
# 設定背景圖像

# background-image:url(leaf.jpg);
```

```
# 不重覆背景圖

# background-repeat:no-repeat;
```

```
# 水平重覆背景圖

# background-repeat:repeat-x;
```

```
# 固定背景圖，不隨頁面滾動

# background-attachment:fixed;
```

```
# 背景圖隨頁面滾動

# background-attachment:scroll;
```

```
# 設定圖片的寬度為100像素，高度為80像素

# background-size:100px 80px;
```

```
# 伸展背景圖完全填充內容區域

# background-size:100% 100%;
```

```
# 背景圖片按比例自適應內容區域

# background-size:contain;
```

```
# 背景簡寫屬性

# background:#fff url(main.png) no-repeat fixed center center;
```

```
# 多重背景

# background-image:url(file1.png),url(file2.png);

# background-position:left top,right bottom;

# background:url(file1.png) left top,url(file2.png) right bottom;
```

```
# 全尺寸背景，用背景圖填滿整個頁面

# background:url(background.png) no-repeat center fixed;background-size:cover;
```

```
# 透明漸變，淡入淡出

# background:linear-gradient(to right, rgba(255,255,255,0), rgba(255,255,255,1));
```

```
# 未指定背景色，預設為透明，覆蓋背景色

# background:url(background.png) no-repeat center fixed;background-color:red;
```
