```
radial-gradient(shape size at position,color1,color2...);	// 徑向漸變語法
```

```
radial-gradient(red,yellow,green);				// 顏色均勻分佈的徑向漸變
```

```
radial-gradient(red 5%,yellow %10,green 20%);	// 顏色依指定比例分佈的徑向漸變
```

```
radial-gradient(circle,red,yellow,green);		// 圓形徑向漸變
```

```
radial-gradient(ellipse,red,yellow,green);		// 楕圓形徑向漸變
```

```
radial-gradient(circle farthest-side at left bottom,red,yellow); 	// 設定漸變結束的大小
```

```
radial-gradient(circle closest-side at left bottom,red,yellow);
```

```
radial-gradient(circle farthest-corner at left bottom,red,yellow);
```

```
radial-gradient(circle closest-corner at left bottom,red,yellow);
```

```
repeating-radial-gradient(red,yellow 10%,green 20%);	// 重覆徑向漸變
```
