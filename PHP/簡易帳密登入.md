PHP
```
$userpass=array(
	'admin'=>'123456',
	'jane'=>'19951022'
);
if( !$_POST['uname'] || $_POST['upass']){
	echo '請輸入帳密';
}else{
	$uname=$_POST['uname'];
	$upass=$_POST['upass'];
	if($userpass[$uname] == $upass){
		echo '登入成功';
	}else{
		echo '帳密不符';
	}
}
```
