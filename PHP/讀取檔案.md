```

$file=file(log.txt);
foreach($file as $line){
	printf('%s<br>',$line);
```

```
$file='log.txt';
```

逐行讀取檔案內容
```
$fp=fopen('log.txt','rb');
while(!feof($fp)){
	printf('%s<br>',fgets($fp));
}
```

逐字讀取檔案內容
```
$fp=fopen('log.txt','rb');
while(!feof($fp)){
	echo fgetc($fp);
}
```

讀取圖片檔
```
header('Content-Type:image/gif');
// 以二進位模式讀取檔案
$open=fopen($file,'rb');
// fread()可以讀取二進位檔(圖片、聲音、影像)
$image=fread($open,filesize($file));
fclose($open);
echo $image;
```
