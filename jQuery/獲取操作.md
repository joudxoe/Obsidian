```
// 獲取h1元素的文本內容
$('h1').text();
```

```
// 獲取h1元素的內容
$('h1').html();
```

```
// 獲取h1元素title屬性的值
$('h1').attr('title');
```

```
// 獲取錨點的連結地址
$('a').prop('href');
```

```
// 獲取id=name的文本輸入的值
$('input[type="text"]#name').val();
```

```
// 獲取所選元素屬性的值
$(selector).css('propertyname');
```

```
// 獲取所選元素的背景色
$(this).css('background-color');
```

```
// 通過索引獲取選擇器匹配的元素
$(selector).get(index);
```

```
// 獲取選擇器匹配的元素相對於同級元素的索引位置
$(selector).index();
```

```
// 獲取元素相對於選擇器的索引位置
$(selector).index(element);
```
