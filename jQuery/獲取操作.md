```
$('h1').text();									// 獲取h1元素的文本內容
```

```
$('h1').html();									// 獲取h1元素的內容
```

```
$('h1').attr('title');							// 獲取h1元素title屬性的值
```

```
$('a').prop('href');							// 獲取錨點的連結地址
```

```
$('input[type="text"]#name').val();				// 獲取id=name的文本輸入的值
```

```
$(selector).css('propertyname');				// 獲取所選元素屬性的值
```

```
$(this).css('background-color');				// 獲取所選元素的背景色
```
