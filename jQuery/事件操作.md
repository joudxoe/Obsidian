```
$(window).keydown(function(event){
	console.log(event.which);						// 通過event.which可以獲取按鍵代碼
});
```

```
$(window).keypress(function(event){
	console.log(event.which);						// 可以獲取輸入字符的ASCII碼
});
```

```
$(window).keypress(function(event){
	console.log(String.fromCharCode(event.which));		// 將輸入的ASCII碼轉換成字符
});
```

```
$(window).keypress(function(event){
	console.log(event.key);							// 獲取鍵盤輸入的字符
});
```

```
$('input[type="text"]').keypress(function(event){
	return event.which>=65 && event.which<=90;		// 文本輸入框只限輸入大寫字母
});
```

```
$('input[type="text"]').keypress(function(event){
	return event.which>=97 && event.which<=122;		// 文本輸入框只限輸入小寫字母
});
```
