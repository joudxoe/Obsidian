```
$ git remote add origin https://github.com/kaochenlong/dummy.git	// 新增遠端數據庫
```

```
$ git remote add origin git@github.com:kaochenlong/dummy.git		// 新增遠端數據庫
```

```
$ git remote		// 顯示所有的遠端數據庫別名
```

```
$ git remote -v		// 檢視所有遠端儲存庫的URL
```

```
$ git remote show https://github.com/joudxoe/Obsidian.git	// 顯示遠程倉庫的信息
```

```
$ git remote rename <old> <new>		// 修改遠端數據庫的名稱
```

```
$ git remote rm <name>		// 刪除遠端儲存庫
```
