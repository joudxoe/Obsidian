```
# 檢視所有的提交記錄
$ git log

# 顯示最近3個提交記錄
$ git log -3

# 倒序顯示所有日誌
$ git log --reverse

# 顯示commit的parent資訊
$ git log --pretty=raw
```

```
# 以簡短一行的方式顯示提交記錄
$ git log --oneline

$ git log --oneline --all

$ git log --oneline --graph
```

```
# 查找版本提交的作者為Jane
$ git log --oneline --author="Jane"

# 查找提交版本的作者為Sam或Jane
$ git log --oneline --author="Sam\|Jane"
```

```
# 查找特定時間範圍的提交記錄
$ git log --since="9am" --until="12pm" --oneline

# 查找特定日期的提交記錄
$ git log --after="2020-10-22" --before="2020-12-31" --oneline
```

```
# 查找包含指定字串內容的提交記錄
$ git log --oneline --grep="WTF"
```

```
# 查找提交記錄的檔案內容
$ git log -S 'Hello, World!'
```

```
# 檢視特定檔案的Commit歷史記錄
$ git log <file>

# 檢視特定檔案每次提交的異動修改記錄
$ git log <file> -p

# 檢視指定目錄下的所有檔案Commit記錄
$ git log <dir> -p
```

```
# 查看指定分支下的commit歷史記錄
$ git log <branch>

# 查看遠端分支的歷史記錄
$ git log remotes/origin/master --oneline
```

```
# 檢視所有commit的reflog
$ git log -g 
```

```
# 隱藏/排除合併提交
$ git log --no-merges

# 只顯示合併提交
$ git log --merges
```

```
# 詳細顯示每個commit的SHA-1值; 查看HEAD的修改記錄
$ git reflog
```

```
# 顯示彙總後的提交記錄
$ git shortlog

$ git shortlog -3
```
