```
# 檢視Git記錄
$ git log

$ git log <branch>

# 以簡短的一行顯示提交記錄
$ git log --oneline

$ git log --oneline --all

$ git log --oneline --graph

# 查看遠端分支的歷史記錄
$ git log remotes/origin/master --oneline
```

```
# 查找版本提交的作者為Jane
$ git log --oneline --author="Jane"

# 查找提交版本的作者為Sam或Jane
$ git log --oneline --author="Sam\|Jane"

# 查找特定時間範圍的提交記錄
$ git log --oneline --since="9am" --until="12pm"
 
# 查找特定日期的提交記錄
$ git log --oneling --after="2020-10"

# 查找包含指定字串內容的提交記錄
$ git log --oneline --grep="WTF"
```

```
# 檢視特定檔案的Commit歷史記錄
$ git log index.php
```

```
# 查看指定分支下的commit歷史記錄
$ git log <branch>
```

```
# 檢視特定檔案每次提交的異動/修改記錄
$ git log -p index.php

# 檢視HTML目錄下的所有檔案Commit記錄
$ git log -p HTML
```

```
# 檢視所有commit的reflog
$ git log -g 
```

```
# 顯示commit的parent資訊
$ git log --pretty=raw
```

```
# 詳細顯示每個commit的SHA-1值; 查看HEAD的修改記錄
$ git reflog
```
